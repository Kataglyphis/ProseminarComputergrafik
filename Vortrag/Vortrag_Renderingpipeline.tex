  \begin{itemize}
    \item nicht beeinflussbar vom Programmierer
    \pause

  \end{itemize}
  \heading{Triangle Traversal}
  \begin{itemize}
    \item alle Dreiecke werden durchgegangen
    \pause
    \item Fragmente werden erzeugt indem Pixel bestimmt werden die innerhalb des Dreiecks liegen
  \end{itemize}
